pipeline {
    agent any
parameters { 
         string(name: 'tomcat_dev', defaultValue: '18.218.209.248', description: 'Staging Server')
        
    } 

    triggers {
         pollSCM('* * * * *') // Polling Source Control
     }
    stages {
        stage('Build') {
            steps {
                sh 'mvn clean package' 
                }
                post {
                  success
                  {
                    echo 'Now Archiving...'
                          archiveArtifacts artifacts: '**/target/*.war'
                }
            }
        }
        stage('Deploy') {
         steps {
                        sh "sudo scp -i 'devo.pem' ./target/*.war ec2-user@${params.tomcat_dev}:/root/apache-tomcat-7.0.53/webapps"
                    }
               }
    }
}
