pipeline {
	agent any
		
	stages{
		stage('Build'){
            		steps {
                		sh 'mvn clean package'
            		}
            		post {
                		success {
                    			echo 'Now Archiving...'
                    			archiveArtifacts artifacts: 'pipe/target/*.war'
                		}
            		}
		}
		stage('deploy'){
			steps{
			sshagent(['admin']){
				sh "scp -o StrictHostKeyChecking=no pipe/target/*.war root@172.31.43.107:/root/apache-tomcat-7.0.53/webapps"

				echo 'deployed..haha'
			}
			}
		}
	}
}
